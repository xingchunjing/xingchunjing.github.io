              # 线程命令

              # 根据pid查看进程

              #ps查询
ps -aux|grep -v grep|grep 3651
#查找进程目录
cd /proc/pid
cwd > 进程目录

              # top命令

              #按进程的CPU使用率排序
运行top命令后，键入大写P

#按进程的内存使用率排序
运行top命令后，键入大写M

              # 查看线程详情

              #查看详情
ps p pid -L -o pcpu,pmem,pid,tid,time,tname,cmd

#查看数量
ps p pid -L -o pcpu,pmem,pid,tid,time,tname,cmd|wc -l

              # 导出堆栈信息

              #导出到当前目录
jstack -l pid > ./jstack.log
